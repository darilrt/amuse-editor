cmake_minimum_required(VERSION 3.5)

# C++ standard\n
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Project name
project(@name)
set(project_name "@name")

# Add definition for shared library
add_compile_definitions(AMUSE_CORE_EXPORT)

# Add source files
file(GLOB_RECURSE source_files assets/components/*.cpp)

# Add library
add_library(
    ${project_name} SHARED
    engine/components.cpp
    ${source_files})

# Include directories
target_include_directories(${project_name} PUBLIC "assets/components")
target_include_directories(${project_name} PUBLIC "engine/core/include")

# Link the library
target_link_directories(${project_name} PUBLIC ${CMAKE_SOURCE_DIR}/engine/core/lib)
target_link_libraries(${project_name} AmuseCore)
